Defining a Custom Filter


nekkidApp.filter('humanScore', function () {
  var map = ['Unrated', 'Rubbish', 'Poor', 'Alright', 'Great', 'Sweeeet!'];
  return function (score) {
    score = parseInt(score, 10);
    return map[score] ? map[score] : score;
  };
});

The Javascript

Return a directive definition object


nekkidApp.directive('myStars', function () {
  return {
    restrict: 'E', // (E)lement, (A)ttribute, (C)lass
    scope: {
      movieScore: '=score' // isolate scope
    },
    templateUrl: 'stars.html'
  };
});

Directive Template


<div class="stars">
  <img src="img/star_{{movieScore >= 1 ? 'on' : 'off'}}.png" ng-click="movieScore = 1">
  <img src="img/star_{{movieScore >= 2 ? 'on' : 'off'}}.png" ng-click="movieScore = 2">
  <img src="img/star_{{movieScore >= 3 ? 'on' : 'off'}}.png" ng-click="movieScore = 3">
  <img src="img/star_{{movieScore >= 4 ? 'on' : 'off'}}.png" ng-click="movieScore = 4">

</div>
